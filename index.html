Strong types vs Weak types


// Java
int number = 7;
number = "Seven"; // Compile error

// Javascript
var number = 7;
number = "Seven"; // No runtime error...

Static typing vs Dynamic types


// Java
Foo foo = new Foo(); // only public method is "bar()"
foo.amfufu(); // COMPILE ERROR!

// Javascript
var foo = new Foo(); // only method as of this line is "bar()"
foo.amfufu(); // RUNTIME ERROR!

Nulls, undefined and truthfullness

Undefined is a primitive that means empty, nil or nothing...

Null is a primitive that means empty, null or nothing...


var a; // interpreted by the JS engine as 'undefined', falsy
var b = null; // falsy
var c = 0; // zero is falsy for some reason, -1 is not :/
var d = ''; // empty string, falsy
var e = "something"; // has value, truthy
var f = {}; // empty object, truthy

console.log(!!a); // false
console.log(!!b); // false
console.log(!!c); // false
console.log(!!d); // false
console.log(!!e); // true
console.log(!!f); // true

Functions expressions (anonymous functions) and callbacks

In JS, functions are first class citizens


function DoThis() {
	// ^ function declaration
}

function() {
	// ^ function expression / anonymous functions
}

var cb = function() {
	console.log("Square result is ");
}

function runCallBackAndSquareX_1(cb, x) {
	cb();
	return x*x;
}

var runCallBackAndSquareX_2 = function(cb, x) {
	cb();
	return x*x;
} 

console.log(runCallBackAndSquareX_1(cb, 7)); // Square result is 49
console.log(runCallBackAndSquareX_2(cb, 6)); // Square result is 36
