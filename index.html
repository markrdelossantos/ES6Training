Nulls, undefined and truthfullness

Undefined is a primitive that means empty, nil or nothing...

Null is a primitive that means empty, null or nothing...


var a; // interpreted by the JS engine as 'undefined', falsy
var b = null; // falsy
var c = 0; // zero is falsy for some reason, -1 is not :/
var d = ''; // empty string, falsy
var e = "something"; // has value, truthy
var f = {}; // empty object, truthy

console.log(!!a); // false
console.log(!!b); // false
console.log(!!c); // false
console.log(!!d); // false
console.log(!!e); // true
console.log(!!f); // true

	console.log("Square result is ");

function runCallBackAndSquareX_1(cb, x) {
	cb();
	return x*x;
}

var runCallBackAndSquareX_2 = function(cb, x) {
	cb();
	return x*x;

ES6 Features

Scoping

Pre-ES6

var is scoped in function block


var outer_variable = 1;

if(true) {
	var if_variable = 2;
}

(function add_one() {
	var sum_variable = 3;
	outer_variable++;
	if_variable++;
	sum_variable++;
})(); // IIFE

console.log(outer_variable); // 2
console.log(if_variable); // 3
console.log(sum_variable); // Reference error

Scoping

ES6

let is scoped in ANY block


var outer_variable = 1;

if(true) {
	let if_variable = 2;
}

(function add_one() {
	var sum_variable = 3;
	outer_variable++;
	if_variable++; // Reference error
	sum_variable++;
})(); // IIFE

console.log(outer_variable); // 2
console.log(if_variable); // Reference error
console.log(sum_variable); // Reference error

Classes

Similar concept to OOP classes


class Duck {
	constructor() {
		this.quackSound = "Quack";
	}

	quack() {
		return `${this.quackSound}!`.repeat(3);
	}
}

let duck = new Duck();
console.log(duck.quack()); // Quack!Quack!Quack!

Classes


class Duck {
	getQuackSound() {
		return "Quack";
	}

	quack() {
		return `${this.quackSound}!`.repeat(3);
	}
}

class RobotDuck extends Duck {
	getQuackSound() {
		return "Meow";
	}

	transform() {
		return "RobotDuck has transformed into a car!";
	}
}

const base_duck = new Duck();
const robo_duck = new RobotDuck();

console.log(base_duck.quack()); // Quack!Quack!Quack!
console.log(robo_duck.quack()); // Meow!Meow!Meow!

console.log(robo_duck.transform()); // RobotDuck has transformed into a car!
console.log(base_duck.transform()); // TypeError: base_duck.transform is not a function
